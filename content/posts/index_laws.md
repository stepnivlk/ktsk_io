$$
a^m \times a^n = a^{m + n} \qquad \frac{a^m}{a^n} = a^{m-n} \\\\
(a^m)^n = a^{mn} \\\\
(a \times b)^n = a^n \times b^n \qquad (\frac{a}{b})^n = \frac{a^n}{b^n} \\\\
a^0 = 1 \qquad a^{-n} = \frac{1}{a^n} \\\\
a^\frac{1}{n} = \sqrt[n]{a} \qquad a^\frac{m}{n} = (\sqrt[n]{a})^m
$$
